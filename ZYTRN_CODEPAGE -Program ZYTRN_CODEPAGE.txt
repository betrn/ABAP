REPORT  ZYTRN_CODEPAGE                .
tables: ztesttrn.
data: ok_code like sy-ucomm.
data: cle like ztesttrn-numero.
*
call screen 100.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
module USER_COMMAND_0100 input.
  case ok_code.
    when 'SAVE'.
      cle = ztesttrn-numero.
      insert ztesttrn.
      set screen 200.
    when others.
      set screen 100.
  endcase.
  clear ok_code.
endmodule.                 " USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*&      Module  READ_ztesttrn  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
module READ_ztesttrn output.
  clear ztesttrn.
  SELECT SINGLE * FROM  ztesttrn CLIENT SPECIFIED
         WHERE  mandt   = sy-mandt
         AND    numero  = cle.

endmodule.                 " READ_ztesttrn  OUTPUT
